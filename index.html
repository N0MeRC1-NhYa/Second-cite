Cousins in Binary Tree


        class Solution {
            public boolean isCousins(TreeNode root, int x, int y) {


         return elDepth(root, x) == elDepth(root, y) && findParent(root, x) != findParent(root, y);

            }
            public int elDepth(TreeNode root, int x){
          if (root == null){
              return 0;
           }
                if (root.val == x){
                    return 50000;
                    }
                int depth = 1 + Math.max(elDepth(root.left, x), elDepth(root.right, x));
            return depth;
                }
            public TreeNode findParent( TreeNode root, int x){
                if (root == null){
                    return null;
                    }
                TreeNode parent = null;
                if ((root.left != null && root.left.val == x) || (root.right != null && root.right.val == x)){
                    parent = root;
                    }
                if ( findParent( root.left, x) != null ){
                    parent = findParent( root.left, x);
                    }
                if ( findParent( root.right, x) != null ){
                    parent = findParent( root.right, x);
                    }
                return parent;
                }
            }
